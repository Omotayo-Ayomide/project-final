// Select elements var ham = document.querySelector(".ham"); var close = document.querySelector(".icon-close"); var firstNav = document.querySelector(".first-nav"); // Hamburger icon click ham.addEventListener("click", () => { firstNav.classList.add("active"); // Add the "active" class ham.style.display = "none"; // Hide the hamburger icon close.style.display = "block"; // Show the close icon }); // Close icon click close.addEventListener("click", () => { firstNav.classList.remove("active"); // Remove the "active" class ham.style.display = "block"; // Show the hamburger icon close.style.display = "none"; // Hide the close icon });
